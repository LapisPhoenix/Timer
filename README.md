# Timer
A Timer CLI


usage: timer.py [-h] [-H HOURS] [-m MINUTES] [-s SECONDS] [-t TITLE] [-d DESCRIPTION]

options:
  -h, --help            show this help message and exit
  -H HOURS, --hours HOURS
  -m MINUTES, --minutes MINUTES
  -s SECONDS, --seconds SECONDS
  -t TITLE, --title TITLE
  -d DESCRIPTION, --description DESCRIPTION
